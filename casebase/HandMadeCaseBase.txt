# Peanut Butter and Jelly
RECIPE BEGIN
NAME
    peanut butter and jelly
INGREDIENTS
    white bread 1
    white bread 2
    peanut butter
    grape jelly
STEPS
    spread(peanut butter, white bread 1)
    stack(white bread 1)
    spread(grape jelly, white bread 2)
    stack(white bread 2)
RESULT
    # The stack is ordered top-to-bottom
    STACK
        white bread 2
        white bread 1
    SPREAD2
        grape jelly
    SPREAD1
        peanut butter
    HOT
    MELTED
    TOASTED
    FRIED
    SLICED
END
